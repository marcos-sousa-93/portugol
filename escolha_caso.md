{
  caso 0:
    // Instruções caso o número for igual a 0
    escreva("O valor é igual a 0")
  pare

  caso 1:
    // Instruções caso o número for igual a 1
    escreva("O valor é igual a 1")
  pare

  caso 2:
    // Instruções caso o número for igual a 2
    escreva("O valor é igual a 2")
  pare

  caso contrario:
    // Instruções caso nenhum dos casos anteriores seja verdadeira
    escreva("O valor não é igual a 0, 1 ou 2")
}
